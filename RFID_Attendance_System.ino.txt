#include <SPI.h>
#include <MFRC522.h>
#include <WiFi.h>
#include <WebServer.h>
#include <time.h>

/* ---------- RFID ---------- */
#define SS_PIN 5
#define RST_PIN 22
MFRC522 mfrc522(SS_PIN, RST_PIN);

/* ---------- LED ----------- */
#define GREEN_LED 26
#define RED_LED   27

/* ---------- WIFI ---------- */
const char* ssid = "sasidhar";
const char* password = "sasi95150";

/* ---------- TIME ---------- */
const char* ntpServer = "pool.ntp.org";
const long gmtOffset_sec = 19800;
const int daylightOffset_sec = 0;

/* ---------- WEB ----------- */
WebServer server(80);
String csvData = "Name,Roll,UID,Date,Time,Status\n";

/* ---------- STUDENTS ------ */
struct Student {
  byte uid[4];
  const char* name;
  int roll;
};

Student students[] = {
  {{0x5C,0xB1,0x31,0x8C}, "SASIDHAR", 29},
  {{0xBA,0xA9,0x2D,0x06}, "SRI LEELA", 18},
  {{0x7C,0xAD,0x0D,0x8D}, "PRASANNA", 34},
  {{0x8C,0x73,0x14,0x8D}, "HARIKA", 33},
  {{0x8C,0x84,0x1C,0x8D}, "SRAVANI", 32}
};
const int TOTAL = sizeof(students) / sizeof(students[0]);

/* ---------- HELPERS ------- */
bool uidMatch(byte *a, byte *b) {
  for (byte i = 0; i < 4; i++) if (a[i] != b[i]) return false;
  return true;
}

String uidHex(byte *u) {
  String s = "";
  for (byte i = 0; i < 4; i++) {
    if (u[i] < 0x10) s += "0";
    s += String(u[i], HEX);
  }
  s.toUpperCase();
  return s;
}

bool getDateTime(String &d, String &t) {
  struct tm ti;
  if (!getLocalTime(&ti)) return false;
  char ds[16], ts[16];
  strftime(ds, sizeof(ds), "%d-%m-%Y", &ti);
  strftime(ts, sizeof(ts), "%H:%M:%S", &ti);
  d = ds; t = ts;
  return true;
}

/* ---------- SETUP --------- */
void setup() {
  Serial.begin(115200);
  SPI.begin();
  mfrc522.PCD_Init();

  pinMode(GREEN_LED, OUTPUT);
  pinMode(RED_LED, OUTPUT);

  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) delay(500);

  configTime(gmtOffset_sec, daylightOffset_sec, ntpServer);
  String d, t;
  while (!getDateTime(d, t)) delay(500);

  Serial.println("IP ADDRESS:");
  Serial.println(WiFi.localIP());

  /* WEB ROUTES */
  server.on("/", []() {
    server.send(200, "text/html",
      "<h2>ESP32 Attendance System</h2>"
      "<a href='/download'>Download Excel</a>");
  });

  server.on("/download", []() {
    server.sendHeader("Content-Type", "text/csv");
    server.sendHeader("Content-Disposition", "attachment; filename=attendance.csv");
    server.send(200, "text/csv", csvData);
  });

  server.begin();
}

/* ---------- LOOP ---------- */
void loop() {
  server.handleClient();

  if (!mfrc522.PICC_IsNewCardPresent()) return;
  if (!mfrc522.PICC_ReadCardSerial()) return;

  byte uid[4];
  for (byte i = 0; i < 4; i++) uid[i] = mfrc522.uid.uidByte[i];

  String name = "UNKNOWN";
  int roll = 0;
  bool granted = false;

  for (int i = 0; i < TOTAL; i++) {
    if (uidMatch(uid, students[i].uid)) {
      name = students[i].name;
      roll = students[i].roll;
      granted = true;
      break;
    }
  }

  String dateStr, timeStr;
  getDateTime(dateStr, timeStr);
  String uidS = uidHex(uid);
  String status = granted ? "GRANTED" : "DENIED";

  digitalWrite(granted ? GREEN_LED : RED_LED, HIGH);
  delay(800);
  digitalWrite(GREEN_LED, LOW);
  digitalWrite(RED_LED, LOW);

  csvData += name + "," + String(roll) + "," + uidS + "," +
             dateStr + "," + timeStr + "," + status + "\n";

  Serial.println(name + " " + status);

  mfrc522.PICC_HaltA();
  mfrc522.PCD_StopCrypto1();
  delay(1500);
}
